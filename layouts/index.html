              int clamp(int value, int min, int max) {
    if (value < min) {
        return min;
    } else if (value > max) {
        return max;
    }
    return value;
}
entry:
  %cmp_min = icmp slt i32 %value, %min
  br i1 %cmp_min, label %ret_min, label %check_max
check_max:
  %cmp_max = icmp sgt i32 %value, %max
  br i1 %cmp_max, label %ret_max, label %ret_value
return_min:
  ret i32 %min
return_max:
  ret i32 %max
return_value:
  ret i32 %value
